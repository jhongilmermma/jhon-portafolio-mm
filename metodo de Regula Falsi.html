def regula_falsi(funcion, a, b, tolerancia, max_iter):
    # Método de la falsa posición para encontrar raíces
    if f(a) * f(b) >= 0:
        return None  # No hay cambio de signo
    
    iteraciones = []
    for i in range(max_iter):
        # Calcular punto de intersección de la cuerda
        fa = f(a)
        fb = f(b)
        c = (a * fb - b * fa) / (fb - fa)
        fc = f(c)
        
        iteraciones.append({'a': a, 'b': b, 'c': c, 'fc': fc})
        
        if abs(fc) < tolerancia:
            return c, iteraciones
        
        # Actualizar intervalo
        if fa * fc < 0:
            b = c
        else:
            a = c
    
    return c, iteraciones
